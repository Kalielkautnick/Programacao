MERGE INTO OF_ITEN_001 OF_ERRADA
USING (
SELECT NUMERO, COR, TAM, SUM(QT_ORIG - (QUANT+QUANT_2+QUANT_I+QUANT_F)) AS QTDE FROM FACCAO_001
WHERE NUMERO = '006579'
AND PARTE = '00'
GROUP BY NUMERO, COR, TAM
) OF_CERTA
ON (OF_ERRADA.NUMERO=OF_CERTA.NUMERO) AND (OF_ERRADA.COR=OF_CERTA.COR) AND (OF_ERRADA.TAM=OF_CERTA.TAM)
WHEN MATCHED THEN 
UPDATE SET OF_ERRADA.QTDE = OF_CERTA.QTDE
